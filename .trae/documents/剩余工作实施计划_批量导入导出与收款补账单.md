```python
@router.put("/{pid}/allocations")
def update_allocations(pid: int, body: dict,
                       user: User = Depends(require_tenant_user),
                       db: Session = Depends(get_db)):
    if not check_permission(user, "payment:verify") and user.role != "tenant_admin":
        raise HTTPException(403, "权限不足")
    q = db.query(PaymentRecord).filter(PaymentRecord.id == pid)
    if user.tenant_id:
        q = q.filter(PaymentRecord.tenant_id == user.tenant_id)
    p = q.first()
    if not p:
        raise HTTPException(404, "收款记录不存在")
    if p.verify_status != "pending":
        raise HTTPException(409, "仅待核对状态可补充账单")
    # 月份锁定校验（同 create_payment）
    lq = db.query(LedgerValidation).filter(
        LedgerValidation.ledger_year == p.payment_date.year,
        LedgerValidation.ledger_month == p.payment_date.month,
        LedgerValidation.status == "locked")
    if user.tenant_id:
        lq = lq.filter(LedgerValidation.tenant_id == user.tenant_id)
    if lq.first():
        raise HTTPException(409, "该月已锁定，无法补充账单")
    # 删除旧分配
    old_allocs = db.query(PaymentBillAllocation).filter(
        PaymentBillAllocation.payment_record_id == pid).all()
    for a in old_allocs:
        db.delete(a)
    # 写入新分配
    total_alloc = Decimal("0")
    for a in body.get("bill_allocations", []):
        amt = Decimal(str(a["allocation_amount"]))
        db.add(PaymentBillAllocation(tenant_id=user.tenant_id, payment_record_id=pid,
                                     bill_id=a["bill_id"], allocation_amount=amt))
        total_alloc += amt
    # 重算预付款：旧分配若小于收款金额，原已加余额，现需还原再加回
    p_amount = Decimal(str(p.amount))
    pq = db.query(CustomerPrepayment).filter(CustomerPrepayment.company_id == p.company_id)
    if user.tenant_id:
        pq = pq.filter(CustomerPrepayment.tenant_id == user.tenant_id)
    prepay = pq.first()
    old_total = sum(a.allocation_amount for a in old_allocs)
    if old_total < p_amount and prepay:
        prepay.balance -= (p_amount - old_total)  # 还原
    if total_alloc < p_amount:
        over = p_amount - total_alloc
        if prepay:
            prepay.balance += over
        else:
            db.add(CustomerPrepayment(tenant_id=user.tenant_id, company_id=p.company_id,
                                      balance=over, source="overpayment"))
    db.commit()
    return {"code": 200, "message": "分配已更新"}
```

**关键点**：
- 仅 `pending` 状态收款可补充账单（已通过/已驳回的不可改）
- 已锁定月份拦截
- 旧分配的预付款余额需先还原，再按新分配重算（避免重复累加）

### 步骤 4：前端通用组件 `ImportExportButtons.vue`

**新增文件**：[frontend/src/components/ImportExportButtons.vue](file:///e:/code/ai_code/payclip/frontend/src/components/ImportExportButtons.vue)

```vue
<template>
  <div class="import-export-buttons" style="display:inline-flex; gap:8px">
    <el-button size="small" @click="onExport">导出 Excel</el-button>
    <el-upload :show-file-list="false" :before-upload="onImport"
               accept=".xlsx,.xls" :action="''">
      <el-button size="small">导入 Excel</el-button>
    </el-upload>
    <el-button size="small" text @click="onTemplate">下载模板</el-button>
  </div>
</template>

<script setup>
import { ElMessage, ElMessageBox } from 'element-plus'
import request from '../utils/request'

const props = defineProps({
  exportUrl: String,      // 如 '/customers/export'
  importUrl: String,      // 如 '/customers/import'
  templateUrl: String,    // 如 '/customers/template'
  exportParams: { type: Object, default: () => ({}) }
})
const emit = defineEmits(['imported'])

async function onExport() {
  const resp = await request.get(props.exportUrl, { params: props.exportParams, responseType: 'blob' })
  triggerDownload(resp, props.exportUrl.split('/').filter(Boolean)[0] + '_export.xlsx')
}

function onImport(file) {
  const fd = new FormData()
  fd.append('file', file)
  request.post(props.importUrl, fd, { headers: { 'Content-Type': 'multipart/form-data' } })
    .then((r) => {
      const d = r.data
      const detail = `成功 ${d.success_count} 条，新建 ${d.created_count} 条，更新 ${d.updated_count} 条，失败 ${d.errors.length} 条`
      const errLines = d.errors.map((e) => `第 ${e.row} 行：${e.reason}`).join('\n')
      ElMessageBox.alert(detail + (errLines ? '\n\n' + errLines : ''), '导入结果', { customClass: 'import-result-box' })
      emit('imported')
    })
  return false  // 阻止 el-upload 默认上传
}

async function onTemplate() {
  const resp = await request.get(props.templateUrl, { responseType: 'blob' })
  triggerDownload(resp, props.templateUrl.split('/').filter(Boolean)[0] + '_template.xlsx')
}

function triggerDownload(resp, filename) {
  const url = URL.createObjectURL(new Blob([resp.data]))
  const a = document.createElement('a')
  a.href = url; a.download = filename; a.click()
  URL.revokeObjectURL(url)
}
</script>
```

**关键点**：
- `onImport` 返回 `false` 阻止 el-upload 默认 XHR，改用全局 `request` 实例（带 Authorization header）
- blob 文件名从 URL 提取（`customers` / `suppliers` / `subscriptions` / `onetime-projects` / `bills`）
- 错误明细逐行展示

3. **script 增加逻辑**：
```js
import { paymentApi, billApi } from '../api'

const allocDialog = ref(false)
const currentPayment = ref(null)
const bills = ref([])
const allocSelected = ref([])

const totalAlloc = computed(() => allocSelected.value.reduce((s, b) => s + Number(b.allocation || 0), 0))
const allocRemainder = computed(() => currentPayment.value ? Math.max(0, Number(currentPayment.value.amount) - totalAlloc.value) : 0)

async function openAllocDialog(row) {
  currentPayment.value = row
  const r = await billApi.list({ company_id: row.company_id, status: 'unpaid', page: 1, page_size: 200 })
  bills.value = r.data.items.map((b) => ({ ...b, allocation: b.receivable_amount - b.paid_amount }))
  // 已选账单默认全选
  allocSelected.value = [...bills.value]
  allocDialog.value = true
}
function onAllocSelect(val) { allocSelected.value = val }

async function saveAllocations() {
  const allocs = allocSelected.value
    .map((b) => ({ bill_id: b.id, allocation_amount: Number(b.allocation || 0) }))
    .filter((a) => a.allocation_amount > 0)
  await paymentApi.updateAllocations(currentPayment.value.id, { bill_allocations: allocs })
  ElMessage.success('账单分配已更新')
  allocDialog.value = false
  load()
}
```
